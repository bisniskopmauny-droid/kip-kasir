function doGet(e) {
  var ss = SpreadsheetApp
    .getActiveSpreadsheet();
  var p = e.parameter;

  // ── helper: balas JSON / JSONP ──
  function reply(obj) {
    var payload = JSON.stringify(obj);
    if (p.callback) {
      return ContentService
        .createTextOutput(
          p.callback+"("+payload+")"
        ).setMimeType(
          ContentService.MimeType.JAVASCRIPT
        );
    }
    return ContentService
      .createTextOutput(payload)
      .setMimeType(
        ContentService.MimeType.JSON
      );
  }

  // ── helper: konversi Date → teks ──
  function toText(v) {
    if (Object.prototype.toString
        .call(v)==="[object Date]") {
      return Utilities.formatDate(
        v,Session.getScriptTimeZone(),
        "yyyy-MM-dd HH:mm");
    }
    return v;
  }

  // ── sheet Data KIP ──────────────
  var sheet = ss.getSheetByName("Data KIP");
  if (!sheet) {
    sheet = ss.insertSheet("Data KIP");
  }
  if (sheet.getLastRow()===0) {
    sheet.appendRow([
      "Nama Talent","Kode",
      "Unit Bisnis","Tanggal",
      "Jam","Bulan"]);
    sheet.getRange("D2:F2000")
      .setNumberFormat("@");
  }

  // ── sheet Talent KIP ────────────
  var tsheet =
    ss.getSheetByName("Talent KIP");
  if (!tsheet) {
    tsheet = ss.insertSheet("Talent KIP");
    tsheet.appendRow(["Nama","Kode"]);
  }

  var action = p.action || "";

  // ── BACA transaksi ──────────────
  if (action==="read") {
    var rows = sheet.getDataRange()
      .getValues();
    rows.shift();
    var data = rows.map(function(r){
      return {
        name:r[0],code:r[1],
        outlet:r[2],
        date:toText(r[3]),
        time:toText(r[4]),
        month:toText(r[5])
      };
    });
    return reply({status:"ok",data:data});
  }

      .getValues();

      .filter(function(r){
        return r[0]&&r[1];
      })
      .map(function(r){
        return {name:r[0],code:r[1]};
      });
    return reply(
      {status:"ok",talents:talents}
    );
