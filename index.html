function doGet(e) {
  var ss = SpreadsheetApp
             .getActiveSpreadsheet();
  var sheet = ss.getSheetByName(
                "Data KIP");
  if (sheet.getLastRow() === 0) {
    sheet.appendRow([
      "Nama Talent","Kode",
      "Unit Bisnis","Tanggal",
      "Jam","Bulan"
    ]);
    // Paksa kolom D:F selalu jadi TEKS,
    // supaya Sheets tidak otomatis
    // mengubahnya jadi tipe Date/Time
    sheet.getRange("D2:F2000")
      .setNumberFormat("@");
  }

  // Jaga-jaga: baca 1 sel apa saja untuk
  // memastikan formatnya teks juga
  function toText(v){
    if (Object.prototype.toString
        .call(v) === "[object Date]") {
      return Utilities.formatDate(
        v, Session.getScriptTimeZone(),
        "yyyy-MM-dd HH:mm"
      );
    }
    return v;
  }

  if (e.parameter.action === "read") {
    var rows = sheet.getDataRange()
                 .getValues();
    rows.shift();
    var data = rows.map(function(r){
      return {
        name:r[0], code:r[1],
        outlet:r[2],
        date:toText(r[3]),
        time:toText(r[4]),
        month:toText(r[5])
      };
    });
    var payload = JSON.stringify(
      {status:"ok", data:data}
    );
    if (e.parameter.callback) {
      return ContentService
        .createTextOutput(
          e.parameter.callback +
          "(" + payload + ")"
        )
        .setMimeType(
          ContentService.MimeType.JAVASCRIPT
        );
    }
    return ContentService
      .createTextOutput(payload)
      .setMimeType(
        ContentService.MimeType.JSON
      );
  }

  sheet.appendRow([
    e.parameter.name   || "",
    e.parameter.code   || "",
    e.parameter.outlet || "",
    "'" + (e.parameter.date  || ""),
    "'" + (e.parameter.time  || ""),
    "'" + (e.parameter.month || "")
  ]);
  return ContentService
    .createTextOutput(
      JSON.stringify({status:"ok"})
    )
    .setMimeType(
      ContentService.MimeType.JSON
    );
}
